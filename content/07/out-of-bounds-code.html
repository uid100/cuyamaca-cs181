Program 8-4:Out-of-bounds code


// Program 8-4: Out-of-bounds code
// This program unsafely stores values beyond an array's boundary.
// What happens depends on how your computer manages memory.
// It MAY overwrite other memory variables. It MAY crash your computer.
// It MAY do nothing at all. It is unpredictable.

#include <iostream>
using namespace std;

int main()
{
    const int SIZE = 3;
    int A[SIZE] = {1, 1, 1}; // Define A as a 3-element int array.
    int B[SIZE];             // Define B as another 3-element int array.

    // Here is what is stored in array A
    cout << "Array A contains: ";
    for (int count = 0; count < SIZE; count++)
        cout << A[count] << "   ";

    // Attempt to store seven numbers in the 3-element array.
    cout << "\n\nAttempting to store 7 numbers in the 3-element array A.\n";
    for (int count = 0; count < 7; count++)
        B[count] = 5; // This is an out-of-bounds error. The result is unpredictable.

    // If the program is still running, display the numbers.
    cout << "\nIf you see this message, the program did not crash."
         << "\n\nHere are the 7 numbers that were stored in the array B:\n";
    for (int count = 0; count < 7; count++)
        cout << B[count] << "   ";
    
    cout << "\n\nHere is what is stored in array A:\n";
    for (int count = 0; count < SIZE; count++)
        cout << A[count] << "   ";

    cout << "\n\nIf you see 5s in array A, the out-of-bounds code overwrote the values in A.\n";

    return 0;
}
